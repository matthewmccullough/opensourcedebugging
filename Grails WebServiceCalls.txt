#Reference Documentation for Grails Web Services
http://grails.org/doc/1.1.x/guide/13.%20Web%20Services.html


#GET ALL SHOWS via REST
curl -i -H "Accept: text/xml" -X GET http://localhost:8080/JackedUpApp2/rest/carShows/
#or with the content type (which really signals what is going in, not out, but it works anyway)
curl -i -H "Content-Type: text/xml" -X GET http://localhost:8080/JackedUpApp2/rest/carShows/
#but not without accept type (REJECTED)
curl -i http://localhost:8080/JackedUpApp2/rest/carShows/
#or a crazy type (ALSO REJECTED)
curl -i -H "Accept: text/garbage" -X GET http://localhost:8080/JackedUpApp2/rest/carShows/

#GET ALL SHOWS via HTML
curl -i http://localhost:8080/JackedUpApp2/carShow/list


#DELETE SHOW 1 via REST
curl -i -H "Accept: text/xml" -X DELETE http://localhost:8080/JackedUpApp2/rest/carShow/1


#GET ONE SHOW via REST
curl -i -H "Accept: text/xml" http://localhost:8080/JackedUpApp2/rest/carShow/1
#or with the content type
curl -i -H "Content-Type: text/xml" http://localhost:8080/JackedUpApp2/rest/carShow/1
#or with JSON
curl -i -H "Content-Type: text/json" http://localhost:8080/JackedUpApp2/rest/carShow/1
but not
curl -i -X GET http://localhost:8080/JackedUpApp2/rest/carShow/1

#CONTENT TYPE DETECTION on HTML CONTROLLER
#GET ONE SHOW VIA HTML
curl -i -X GET http://localhost:8080/JackedUpApp2/carShow/show/1
curl -i -H "Accept: text/xml" http://localhost:8080/JackedUpApp2/carShow/show/1


#SAVE A NEW SHOW
curl -i -H "Content-Type: text/xml" -X POST http://localhost:8080/JackedUpApp2/rest/carShow/ --data "<carShow><city>NEWNEW</city><minimumCompetitors>6</minimumCompetitors><state>NEW</state><venue>NEW</venue></carShow>"
#but you can't use accept type, as it only controls the READER methods, not the SAVING ones, or it doesn't parse the XML
curl -i -H "Accept: text/xml" -X POST http://localhost:8080/JackedUpApp2/rest/carShow/ --data "<carShow><city>NEWNEW</city><minimumCompetitors>6</minimumCompetitors><state>NEW</state><venue>NEW</venue></carShow>"


#UPDATE A SHOW
curl -i -H "Content-Type: text/xml" -X PUT http://localhost:8080/JackedUpApp2/rest/carShow/1 --data "<carShow id=\"1\"><city>UPDATED</city><minimumCompetitors>6</minimumCompetitors><state>UPDATED</state><venue>UPDATED</venue></carShow>"
